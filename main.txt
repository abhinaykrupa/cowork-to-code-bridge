#!/bin/bash

# docker_logs.sh - Tail logs of a Docker container
# Usage: docker_logs.sh <container_name_or_id> [line_count]
# Default line count: 50

set -euo pipefail

# Color codes for output
readonly RED='\033[0;31m'
readonly GREEN='\033[0;32m'
readonly YELLOW='\033[1;33m'
readonly NC='\033[0m' # No Color

# Function to print error messages
error_exit() {
    echo -e "${RED}Error: $1${NC}" >&2
    exit 1
}

# Function to print usage information
usage() {
    echo "Usage: $0 <container_name_or_id> [line_count]"
    echo "  container_name_or_id - Name or ID of the Docker container"
    echo "  line_count           - Number of lines to tail (default: 50)"
    echo ""
    echo "Examples:"
    echo "  $0 my_container"
    echo "  $0 my_container 100"
    echo "  $0 abc123def 20"
    exit 1
}

# Function to check if Docker daemon is running
check_docker_running() {
    if ! docker info &>/dev/null; then
        error_exit "Docker daemon is not running. Please start Docker first."
    fi
}

# Function to check if container exists
check_container_exists() {
    local container="$1"
    if ! docker ps -a --format '{{.Names}} {{.ID}}' | grep -qi "$container"; then
        error_exit "Container '$container' does not exist. Use 'docker ps -a' to list available containers."
    fi
}

# Main function
main() {
    # Check if at least one argument is provided
    if [[ $# -lt 1 ]]; then
        usage
    fi

    local container="$1"
    local line_count="${2:-50}"

    # Validate line count is a positive integer
    if ! [[ "$line_count" =~ ^[0-9]+$ ]] || [[ "$line_count" -eq 0 ]]; then
        error_exit "Line count must be a positive integer."
    fi

    # Check Docker daemon
    check_docker_running

    # Check container existence
    check_container_exists "$container"

    # Tail the logs
    echo -e "${GREEN}Tailing last $line_count lines of container '$container':${NC}"
    docker logs --tail "$line_count" -f "$container" 2>&1 || {
        local exit_code=$?
        if [[ $exit_code -eq 125 ]]; then
            error_exit "Failed to access container logs. Container may be in a bad state."
        elif [[ $exit_code -eq 126 ]]; then
            error_exit "Permission denied. Try running with sudo or check Docker permissions."
        else
            error_exit "Failed to tail logs (exit code: $exit_code)."
        fi
    }
}

# Run main function with all arguments
main "$@"
